Εντολές για Styling

Αλλαγή Χρώματος

document.getElementById('text').style.color = 'blue'; //Αλλαγή Χρώματος Κειμένου
document.getElementById('box').style.backgroundColor = 'green'; //Αλλαγή Χρώματος Φόντου

Αλλαγή Μεγέθους Γραμματοσειράς

document.getElementById('text').style.fontSize = '24px';

Προσθήκη Περιγράμματος

document.getElementById('box').style.border = '2px solid black';

Αλλαγή Ύψους και Πλάτους

document.getElementById('box').style.width = '200px';
document.getElementById('box').style.height = '150px';

Κεντρική Στοίχιση Κειμένου

document.getElementById('text').style.textAlign = 'center';

Εφαρμογή Σκιάς Κειμένου

document.getElementById('text').style.textShadow = '2px 2px 4px #000000';

Στρογγυλοποίηση Γωνιών

document.getElementById('box').style.borderRadius = '15px';

Εμφάνιση και Απόκρυψη Στοιχείου

// Για να κρύψετε το στοιχείο
document.getElementById('box').style.display = 'none';
// Για να εμφανίσετε το στοιχείο
document.getElementById('box').style.display = 'block';

Αλλαγή Πάχους Γραμμής (Line Height)

document.getElementById('text').style.lineHeight = '1.5';

Αλλαγή Σκηνής (Opacity)

document.getElementById('box').style.opacity = '0.5'; // Απόλυτη διαφάνεια 50%

Εφαρμογή Ριμπαριστού Χρώματος (Gradient Background)

document.getElementById('box').style.background = 'linear-gradient(to right, red, yellow)';

Εφαρμογή Εφέ Hover (Με JavaScript)

document.getElementById('box').addEventListener('mouseover', function() {
    this.style.transform = 'scale(1.1)'; // Μεγέθυνση στο hover
});
document.getElementById('box').addEventListener('mouseout', function() {
    this.style.transform = 'scale(1)'; // Επαναφορά στο κανονικό μέγεθος
});

Αλλαγή Εφαρμογής Σκιάς Κουτιού (Box Shadow)

document.getElementById('box').style.boxShadow = '5px 5px 10px #000000';

Ρύθμιση Χώρου Εσωτερικά (Padding)

document.getElementById('box').style.padding = '20px';

Ρύθμιση Χώρου Εξωτερικά (Margin)

document.getElementById('box').style.margin = '15px';

Εφαρμογή Εφέ Περικοπής (Clip Path)

document.getElementById('box').style.clipPath = 'circle(50%)'; // Κυκλική περικοπή

Εφαρμογή Εφέ Σκηνής (Filter)

document.getElementById('image').style.filter = 'grayscale(100%)'; // Σκοτεινία της εικόνας

Ρύθμιση Εύρους Περιγράμματος (Border Width)

document.getElementById('box').style.borderWidth = '5px';

Αλλαγή Στυλ Περιγράμματος (Border Style)

document.getElementById('box').style.borderStyle = 'dashed'; // Διακεκομμένο περίγραμμα

Εφαρμογή Εφέ Κίνησης (Transform)

document.getElementById('box').style.transform = 'rotate(45deg)'; // Περιστροφή 45 μοίρες

Εφαρμογή Σκηνής με CSS Variables

document.documentElement.style.setProperty('--main-bg-color', 'lightblue');

Εφαρμογή Κινητικών Εφέ με CSS Animations

// Ορίστε μια animation μέσω CSS πρώτα
// @keyframes example { 0% { transform: translateX(0); } 100% { transform: translateX(100px); } }
// .animated { animation: example 2s ease-in-out; }
document.getElementById('box').classList.add('animated');

Δημιουργία Εφέ Υποβάθρου με Multiple Backgrounds

document.getElementById('box').style.background = 'url(image.jpg), linear-gradient(to right, red, blue)';

Εφαρμογή 3D Transformation

document.getElementById('box').style.transform = 'rotateX(30deg) rotateY(45deg)';
document.getElementById('box').style.transformStyle = 'preserve-3d'; // Απαιτείται για 3D μετασχηματισμούς

Δημιουργία Εφέ CSS Clip Path με Πολύγωνα

document.getElementById('box').style.clipPath = 'polygon(50% 0%, 100% 50%, 50% 100%, 0% 50%)';

Διαχείριση Διαφορετικών Media Queries με JavaScript

if (window.matchMedia('(min-width: 600px)').matches) {
    document.getElementById('box').style.backgroundColor = 'lightgreen';
} else {
    document.getElementById('box').style.backgroundColor = 'lightcoral';
}

Εφαρμογή Εφέ CSS Transitions

document.getElementById('box').style.transition = 'background-color 0.5s ease';
document.getElementById('box').onmouseover = function() {
    this.style.backgroundColor = 'orange';
};
document.getElementById('box').onmouseout = function() {
    this.style.backgroundColor = 'lightblue';
};

Ρύθμιση Grid Layout Properties

const container = document.getElementById('grid-container');
container.style.display = 'grid';
container.style.gridTemplateColumns = 'repeat(3, 1fr)';
container.style.gridGap = '10px';

Δημιουργία Κινητικών Εφέ με CSS Transforms

document.getElementById('box').style.transform = 'scale(1.5) translateX(50px)';
document.getElementById('box').style.transition = 'transform 0.3s ease';

Εφαρμογή Custom Scrollbars

document.getElementById('box').style.overflow = 'auto';
document.getElementById('box').style.scrollbarWidth = 'thin'; // Εφαρμογή λεπτών scrollbars (Firefox)
